def calculate_bill(units):
    # Initialize charges
    first_charge = second_charge = remaining_charge = 0

    # First 100 units → ₹2/unit
    if units > 0:
        first_units = min(units, 100)
        first_charge = first_units * 2

    # Next 100 units → ₹3/unit
    if units > 100:
        second_units = min(units - 100, 100)
        second_charge = second_units * 3

    # Remaining units → ₹5/unit
    if units > 200:
        remaining_units = units - 200
        remaining_charge = remaining_units * 5
    else:
        remaining_units = 0

    # Total bill
    total = first_charge + second_charge + remaining_charge

    # Display breakdown
    print(f"Units = {units}")
    print(f"First 100 units = 100 * 2 = {first_charge}")
    print(f"Second 100 units = 100 * 3 = {second_charge}")
    print(f"Remaining {remaining_units} units = {remaining_units} * 5 = {remaining_charge}")
    print(f"Total Bill = {total}")


# Take input manually
units = int(input("Enter number of units consumed: "))
calculate_bill(units)






 
# ATM Withdrawal Program

def atm_transaction():
    # Initial balance stored in the system
    balance = 5000.00
    print(f"Current Balance: ₹{balance}")

    try:
        # Taking withdrawal amount as input
        amount = int(input("Enter the amount to withdraw (multiples of 100): "))

        # Validation 1: Check if amount is a multiple of 100
        if amount % 100 != 0:
            print("Error: Amount must be a multiple of 100 (e.g., 100, 200, 500).")
        
        # Validation 2: Check for sufficient balance
        elif amount > balance:
            print("Error: Insufficient balance.")
        
        # Success: Deduct and update
        else:
            balance -= amount
            print("Transaction successful!")
            print(f"Please collect your cash. Updated Balance: ₹{balance}")

    except ValueError:
        print("Error: Please enter a valid whole number.")

# Run the program
atm_transaction()
